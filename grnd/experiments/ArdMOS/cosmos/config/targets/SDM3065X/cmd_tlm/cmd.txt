COMMAND SDM3065X SCAN_SET_UP BIG_ENDIAN "Do set up to start scanning"
  APPEND_ID_PARAMETER CMD_ID 8 UINT 1 1 1 "Command Id" # Unique command ID
  APPEND_PARAMETER CMD_TEMPLATE 512 STRING ":ROUT:SCAN ON; :ROUT:FUNC STEP;:ROUT:DCV:AZ OFF;"
 
COMMAND SDM3065X TURN_ON_CHANNEL BIG_ENDIAN "Turn on channel for scanning"
  APPEND_ID_PARAMETER CMD_ID 8 UINT 1 1 1 "Command Id" # Unique command ID
  APPEND_PARAMETER CHANNEL 512 STRING "channel"
  APPEND_PARAMETER CMD_TEMPLATE 512 STRING "ROUT: CHAN <channel>,ON,DCV,AUTO,FAST"

COMMAND SDM3065X SET_SCANNING_RANGE BIG_ENDIAN "set the range of channels to scan"
  APPEND_ID_PARAMETER CMD_ID 8 UINT 1 1 1 "Command Id" # Unique command ID
  APPEND_PARAMETER low 512 STRING "low"
  APPEND_PARAMETER high 512 STRING "high"
  APPEND_PARAMETER CMD_TEMPLATE 512 STRING ":ROUTe:LIMI:HIGH <high>;:ROUTe:LIMI:LOW <low>"
  
COMMAND SDM3065X START_SCAN BIG_ENDIAN "Start scanning through the channels"
  APPEND_ID_PARAMETER CMD_ID 8 UINT 1 1 1 "Command Id" # Unique command ID
  APPEND_PARAMETER CMD_TEMPLATE 512 STRING ":ROUT:START ON"
 
COMMAND SDM3065X GET_DATA BIG_ENDIAN "Get the data from a channel."
  APPEND_ID_PARAMETER CMD_ID 8 UINT 1 1 1 "Command Id" # Unique command ID
  APPEND_PARAMETER CHANNEL 512 STRING "channel"
  APPEND_PARAMETER CMD_TEMPLATE 512 STRING ":ROUT:DATA? <CHANNEL>"
  APPEND_PARAMETER RSP_TEMPLATE 512 STRING "<DATA>"
  APPEND_PARAMETER RSP_PACKET 512 STRING "DATA"
  
COMMAND SDM3065X END_SCAN BIG_ENDIAN "Stop scanning through the channels"
  APPEND_ID_PARAMETER CMD_ID 8 UINT 1 1 1 "Command Id" # Unique command ID
  APPEND_PARAMETER CMD_TEMPLATE 512 STRING ":ROUT:START OFF"
